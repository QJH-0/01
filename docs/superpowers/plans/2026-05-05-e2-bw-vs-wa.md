# D:\Paper\01\src\data\librimix.py
from __future__ import annotations

from typing import Any

import torch
from torch.utils.data import DataLoader
from torchaudio.datasets import LibriMix


def _collate_librimix(batch: list[tuple[int, torch.Tensor, list[torch.Tensor]]]):
    mixes = []
    sources = []
    for _, mix, srcs in batch:
        mixes.append(mix)
        sources.append(torch.cat(srcs, dim=0))
    return torch.stack(mixes, dim=0), torch.stack(sources, dim=0)


def _make_dataset(root: str, subset: str, task: str, sample_rate: int, mode: str, num_speakers: int):
    return LibriMix(
        root=root,
        subset=subset,
        num_speakers=num_speakers,
        sample_rate=sample_rate,
        task=task,
        mode=mode,
    )


def build_dataloaders(config: dict[str, Any]):
    train_set = _make_dataset(
        root=config["root"],
        subset=config["train_subset"],
        task=config["task"],
        sample_rate=config["sample_rate"],
        mode=config["mode"],
        num_speakers=config["num_speakers"],
    )
    valid_set = _make_dataset(
        root=config["root"],
        subset=config["valid_subset"],
        task=config["task"],
        sample_rate=config["sample_rate"],
        mode=config["mode"],
        num_speakers=config["num_speakers"],
    )
    common = dict(
        batch_size=config["batch_size"],
        num_workers=config["num_workers"],
        pin_memory=config["pin_memory"],
        collate_fn=_collate_librimix,
    )
    train_loader = DataLoader(train_set, shuffle=True, persistent_workers=config["num_workers"] > 0, **common)
    valid_loader = DataLoader(valid_set, shuffle=False, persistent_workers=False, **common)
    return train_loader, valid_loader
```

# D:\Paper\01\src\models\binarize.py
from __future__ import annotations

import torch
import torch.nn as nn
import torch.nn.functional as F


class SignSTE(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x):
        return x.sign()

    @staticmethod
    def backward(ctx, grad_output):
        return grad_output


def ste_sign(x: torch.Tensor) -> torch.Tensor:
    return SignSTE.apply(x)


def binarize_weight(weight: torch.Tensor) -> torch.Tensor:
    alpha = weight.abs().mean(dim=(1, 2), keepdim=True)
    binary = alpha * ste_sign(weight)
    return binary.detach() + weight - weight.detach()


class BinaryConv1d(nn.Module):
    def __init__(self, conv: nn.Conv1d):
        super().__init__()
        self.weight = conv.weight
        self.bias = conv.bias
        self.stride = conv.stride
        self.padding = conv.padding
        self.dilation = conv.dilation
        self.groups = conv.groups

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return F.conv1d(
            x,
            binarize_weight(self.weight),
            self.bias,
            stride=self.stride,
            padding=self.padding,
            dilation=self.dilation,
            groups=self.groups,
        )


class BinaryActivation(nn.Module):
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return ste_sign(x)
```

# D:\Paper\01\src\models\patch_tcn.py
from __future__ import annotations

import copy
import torch.nn as nn

from src.models.binarize import BinaryActivation, BinaryConv1d


def _replace_shared_block(shared_block: nn.Sequential, use_binary_activation: bool) -> nn.Sequential:
    layers = []
    for layer in shared_block:
        if isinstance(layer, nn.Conv1d):
            layers.append(BinaryConv1d(layer))
        elif isinstance(layer, nn.PReLU) and use_binary_activation:
            layers.append(BinaryActivation())
        else:
            layers.append(layer)
    return nn.Sequential(*layers)


def _patch_conv_block(block: nn.Module, use_binary_activation: bool) -> nn.Module:
    block.shared_block = _replace_shared_block(block.shared_block, use_binary_activation)
    block.res_conv = BinaryConv1d(block.res_conv)
    if hasattr(block, "skip_conv"):
        block.skip_conv = BinaryConv1d(block.skip_conv)
    return block


def apply_e2_mode(model: nn.Module, mode: str) -> nn.Module:
    if mode == "fp32_baseline":
        return model
    if mode not in {"bw_tcn", "wa_tcn"}:
        raise ValueError(f"Unsupported E2 mode: {mode}")
    use_binary_activation = mode == "wa_tcn"
    for idx, block in enumerate(model.masker.TCN):
        model.masker.TCN[idx] = _patch_conv_block(block, use_binary_activation)
    return model
```

# D:\Paper\01\src\train\losses.py
from __future__ import annotations

import torch


def si_snr(estimate: torch.Tensor, target: torch.Tensor, eps: float = 1e-8) -> torch.Tensor:
    estimate = estimate - estimate.mean(dim=-1, keepdim=True)
    target = target - target.mean(dim=-1, keepdim=True)
    proj = (torch.sum(estimate * target, dim=-1, keepdim=True) / (torch.sum(target ** 2, dim=-1, keepdim=True) + eps)) * target
    noise = estimate - proj
    ratio = (torch.sum(proj ** 2, dim=-1) + eps) / (torch.sum(noise ** 2, dim=-1) + eps)
    return 10 * torch.log10(ratio + eps)


def pit_si_snr_loss(estimate: torch.Tensor, target: torch.Tensor) -> torch.Tensor:
    est_a, est_b = estimate[:, 0], estimate[:, 1]
    tgt_a, tgt_b = target[:, 0], target[:, 1]
    pair_1 = si_snr(est_a, tgt_a) + si_snr(est_b, tgt_b)
    pair_2 = si_snr(est_a, tgt_b) + si_snr(est_b, tgt_a)
    best = torch.maximum(pair_1, pair_2)
    return -best.mean()


def batch_si_snri(estimate: torch.Tensor, target: torch.Tensor, mixture: torch.Tensor) -> float:
    mix = mixture.squeeze(1)
    baseline = (si_snr(mix, target[:, 0]) + si_snr(mix, target[:, 1])) / 2
    est_score = torch.maximum(
        (si_snr(estimate[:, 0], target[:, 0]) + si_snr(estimate[:, 1], target[:, 1])) / 2,
        (si_snr(estimate[:, 0], target[:, 1]) + si_snr(estimate[:, 1], target[:, 0])) / 2,
    )
    return float((est_score - baseline).mean().item())
```

```python
# D:\Paper\01\src\train\engine.py
from __future__ import annotations

from pathlib import Path

import torch
from torch import optim
from torch.amp import GradScaler, autocast

from src.data.librimix import build_dataloaders
from src.models.patch_tcn import apply_e2_mode
from src.models.pretrained import load_pretrained_convtasnet
from src.train.losses import batch_si_snri, pit_si_snr_loss
from src.utils.io import ensure_dir, load_yaml, write_json
from src.utils.seed import set_seed


def _resolve_device(name: str) -> torch.device:
    if name == "cuda" and torch.cuda.is_available():
        return torch.device("cuda")
    return torch.device("cpu")


def _forward_loss(model, mixture, target):
    estimate = model(mixture)
    loss = pit_si_snr_loss(estimate, target)
    return estimate, loss


def run_one_debug_step(config_path: Path):
    config = load_yaml(config_path)
    set_seed(config["experiment"]["seed"])
    device = _resolve_device(config["train"]["device"])
    train_loader, _ = build_dataloaders(config["data"])
    mixture, target = next(iter(train_loader))
    mixture = mixture.to(device)
    target = target.to(device)
    model = load_pretrained_convtasnet(config["model"]["pretrained_name"]).to(device)
    model = apply_e2_mode(model, config["experiment"]["mode"])
    optimizer = optim.AdamW(model.parameters(), lr=config["train"]["lr"], weight_decay=config["train"]["weight_decay"])
    scaler = GradScaler(device="cuda", enabled=config["train"]["amp"] and device.type == "cuda")
    optimizer.zero_grad(set_to_none=True)
    with autocast(device_type=device.type, enabled=config["train"]["amp"] and device.type == "cuda"):
        estimate, loss = _forward_loss(model, mixture, target)
    scaler.scale(loss).backward()
    scaler.step(optimizer)
    scaler.update()
    return {"loss": float(loss.item()), "si_snri": batch_si_snri(estimate.detach(), target, mixture)}


def run_experiment(config_path: Path):
    config = load_yaml(config_path)
    set_seed(config["experiment"]["seed"])
    device = _resolve_device(config["train"]["device"])
    output_dir = ensure_dir(Path(config["experiment"]["output_dir"]))
    train_loader, valid_loader = build_dataloaders(config["data"])
    model = load_pretrained_convtasnet(config["model"]["pretrained_name"]).to(device)
    model = apply_e2_mode(model, config["experiment"]["mode"])
    optimizer = optim.AdamW(model.parameters(), lr=config["train"]["lr"], weight_decay=config["train"]["weight_decay"])
    scaler = GradScaler(device="cuda", enabled=config["train"]["amp"] and device.type == "cuda")
    best = {"best_val_loss": float("inf")}

    for epoch in range(config["train"]["epochs"]):
        model.train()
        train_losses = []
        for mixture, target in train_loader:
            mixture = mixture.to(device)
            target = target.to(device)
            optimizer.zero_grad(set_to_none=True)
            with autocast(device_type=device.type, enabled=config["train"]["amp"] and device.type == "cuda"):
                _, loss = _forward_loss(model, mixture, target)
            scaler.scale(loss).backward()
            scaler.step(optimizer)
            scaler.update()
            train_losses.append(float(loss.item()))

        model.eval()
        valid_losses = []
        valid_snri = []
        with torch.no_grad():
            for mixture, target in valid_loader:
                mixture = mixture.to(device)
                target = target.to(device)
                estimate, loss = _forward_loss(model, mixture, target)
                valid_losses.append(float(loss.item()))
                valid_snri.append(batch_si_snri(estimate, target, mixture))

        epoch_metrics = {
            "epoch": epoch + 1,
            "train_loss": sum(train_losses) / len(train_losses),
            "val_loss": sum(valid_losses) / len(valid_losses),
            "val_si_snri": sum(valid_snri) / len(valid_snri),
        }
        if epoch_metrics["val_loss"] < best["best_val_loss"]:
            best = dict(epoch_metrics)
            torch.save(model.state_dict(), output_dir / "best_model.pt")

    write_json(output_dir / "summary.json", best)
    return best
```
